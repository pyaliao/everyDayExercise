### 写一个方法去掉字符串中的空格

```js

str.split(' ').join('')

str.replace(/\s/g, '')

str.replaceAll(' ', '')

const delSpace = function (str, pos='all') {
  if (!str) {
    return str
  }
  const patterns = {
    'all': {
      pt: /\s/g
    },
    'left': {
      pt: /^\s+/g
    },
    'right': {
      pt: /\s+$/g
    },
    'center': {
      pt: /(\S+)(\s+)(\S+)/g,
      grp: '$1$3'
    },
    'both': {
      pt: /(^\s+)|(\s+$)/g
    }
  }
  return str.replace(patterns[pos].pt, patterns[pos].grp ? patterns[pos].grp : '')
}
```
